3   public class StopWatch {
4   
5       private long beginTime;
6       private final String message;
7   
8       public StopWatch() {
9           this("");
10      }
11  
12      public StopWatch(String message) {
13          reset();
14          this.message = message;
15      }
16  
17      public void reset() {
18          this.beginTime = System.currentTimeMillis();
19      }
20  
21      /**
22       * Get time since {@code this} instance was created or since last {@link #reset()}.
23       *
24       * @return formatted time like dd/hh/mm/ss/sss
25       */
26      public String getFormattedTime() {
27          final String readableFormat = "%02d:%02d:%02d.%03d";
28          long millis = System.currentTimeMillis() - beginTime;
29          int sec = (int) (millis / 1000) % 60;
30          int min = (int) ((millis / (1000 * 60)) % 60);
31          int hours = (int) ((millis / (1000 * 60 * 60)) % 24);
32          return String.format(readableFormat, hours, min, sec, millis % 1000);
33      }
34  
35      public String getMessage() {
36          return message;
37      }
38  
39      @Override
40      public String toString() {
41          return message + getFormattedTime();
42      }
43  }
