123     public List<Vertex> generateDAG() {
124         List<Vertex> vertices = new ArrayList<>();
125 
126         // Create vertices
127         for (int i = 0; i < numVertices; i++) {
128             Vertex vertex = new Vertex("V" + i);
129             vertices.add(vertex);
130         }
131 
132         for (int i = 0; i < numVertices - 1; i++) {
133             Vertex startVertex = vertices.get(i);
134             for (int j = i + 1; j - i - 1 < numEdges && j < numVertices; j++) {
135                 final Vertex nextVertex = vertices.get(j);
136                 link(RANDOM_WEIGHTS[(j - 1) % RANDOM_WEIGHTS.length], startVertex, nextVertex);
137             }
138         }
139 
140         return vertices;
141     }

143     private static void link(double weight, Vertex startVertex, Vertex targetVertex) {
144         // Check if there is already an edge from startVertex to targetVertex
145         boolean edgeExists = startVertex.getAdjacencyList().stream()
146             .anyMatch(edge -> edge.getTargetVertex().equals(targetVertex));
147 
148         if (!edgeExists && !startVertex.equals(targetVertex)) {
149             Edge edge = new Edge(weight, startVertex, targetVertex);
150             startVertex.addNeighbor(edge);
151         }
152     }

156 class DijkstraAlgorithm {
157     private final Function<Object, Object> poll;
158     private final Predicate<Object> isEmpty;
159     private final BiConsumer<Object, Object> push;
160     private final BiConsumer<Object, Object> use;
161     private final Object dataStructure;
162 
163     public DijkstraAlgorithm(Function<Object, Object> poll, BiConsumer<Object, Object> push,
164                              BiConsumer<Object, Object> use, Object dataStructure, Predicate<Object> isEmpty) {
165         this.poll = poll;
166         this.push = push;
167         this.use = use;
168         this.dataStructure = dataStructure;
169         this.isEmpty = isEmpty;
170     }
171 
172     public static <H, V> DijkstraAlgorithm createDefault(Function<H, Object> poll, BiConsumer<H, V> push,
173                                                          BiConsumer<H, V> use, H dataStructure, Predicate<H> isEmpty) {
174         return new DijkstraAlgorithm(convert(poll), convert(push), convert(use), dataStructure, convert(isEmpty));
175     }
176 
177     @SuppressWarnings("unchecked")
178     private static <H, V> BiConsumer<Object, Object> convert(BiConsumer<H, V> c) {
179         return (a, b) -> c.accept((H) a, (V) b);
180     }
181 
182     @SuppressWarnings("unchecked")
183     private static <H> Function<Object, Object> convert(Function<H, Object> op) {
184         return a -> op.apply((H) a);
185     }
186 
187     @SuppressWarnings("unchecked")
188     private static <H> Predicate<Object> convert(Predicate<H> op) {
189         return a -> op.test((H) a);
190     }
191 
192     public void computePath(Vertex source) {
193         // heap
194         push.accept(dataStructure, source);
195         source.setDistance(0);
196         while (!isEmpty.test(dataStructure)) {
197             final Vertex actualVertex = (Vertex) poll.apply(dataStructure);
198             for (Edge edge : actualVertex.getAdjacencyList()) {
199                 final Vertex v = edge.getTargetVertex();
200                 final double d = actualVertex.getDistance() + edge.getWeight();
201                 if (d < v.getDistance()) {
202                     v.setDistance(d);
203                     v.setPredecessor(actualVertex);
204                     use.accept(dataStructure, v);
205                 }
206             }
207         }
208     }
209 
210     public double getShortestPathDistanceTo(Vertex targetVertex) {
211         double sum = 0;
212         for (Vertex vertex = targetVertex; vertex != null; vertex = vertex.getPredecessor()) {
213             sum += vertex.getDistance();
214         }
215         return sum;
216     }
