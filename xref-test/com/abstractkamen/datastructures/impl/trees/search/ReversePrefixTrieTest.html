129     @Test
130     public void testDelete() {
131         final SuffixTrie tree = new ReversePrefixTrie();
132         tree.insert("app");
133         tree.insert("apple");
134         tree.insert("application");
135         tree.insert("banana");
136         assertEquals(42, tree.size());
137         assertEquals(4, tree.completeWords());
138         assertTrue(tree.delete("apple"));
139         assertFalse(tree.delete("orange"));
140         assertEquals(36, tree.size());
141         assertEquals(3, tree.completeWords());
142         final Collection<String> startsWithResult = tree.startsWith("app", Integer.MAX_VALUE);
143         assertEquals(startsWithResult, List.of("app", "application"));
144         tree.insert("cat");
145         tree.insert("car");
146         assertEquals(46, tree.size());
147         assertEquals(5, tree.completeWords());
148         assertTrue(tree.delete("cat"));
149         assertEquals(42, tree.size());
150         assertEquals(4, tree.completeWords());
151         assertTrue(tree.delete("app"));
152         assertEquals(39, tree.size());
153         assertEquals(3, tree.completeWords());
154         System.out.println(tree.prettyString());
155         assertTrue(tree.delete("application"));
156         System.out.println(tree.prettyString());
157         assertEquals(18, tree.size());
158         assertEquals(2, tree.completeWords());
159         assertTrue(tree.delete("banana"));
160         System.out.println(tree.prettyString());
161         assertEquals(6, tree.size());
162         assertEquals(1, tree.completeWords());
163         assertTrue(tree.delete("car"));
164         final String actualEmpty = tree.prettyString();
165         assertEquals("empty tree", actualEmpty);
166         System.out.println(actualEmpty);
167         assertEquals(0, tree.size());
168         assertEquals(0, tree.completeWords());
169     }
