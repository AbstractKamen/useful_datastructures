13  /**
14   * Demo class showing PrefixTrie features and behaviour.
15   */
16  public class PrefixTrieDemo {
17      public static void main(String[] args) {
18          final Collection<String> list = List.of("apple", "apricot", "avocado",
19                                                  "ant", "app", "ants", "apps", "apis",
20                                                  "analyse", "alms", "alarm", "alarms");
21          final PrefixTrie trie = new PrefixTrieImpl();
22          list.forEach(trie::insert);
23          System.out.println(trie);
24          System.out.println(trie.prettyString());
25  
26          startsWithDemo(trie, "a");
27          startsWithDemo(trie, "ap");
28          startsWithDemo(trie, "app");
29          startsWithDemo(trie, "apple");
30          deleteDemo(trie, "appi");
31          deleteDemo(trie, "app");
32          startsWithDemo(trie, "app");
33          deleteDemo(trie, "api");
34          startsWithDemo(trie, "a");
35          System.out.println(trie.prettyString());
36          list.forEach(s -> deleteDemo(trie, s));
37          System.out.println(trie);
38          System.out.println(trie.prettyString());
39          startsWithDemo(trie, "a");
40          System.out.println("size: " + trie.size());
41          System.out.println("words: " + trie.completeWords());
42  
43          demoShakespeare();
44      }
45  
46      private static void demoShakespeare() {
47          try (final InputStream inputStream = PrefixTrieDemo.class.getClassLoader().getResourceAsStream("shakespeare.txt")) {
48              final BufferedReader reader = new BufferedReader(new InputStreamReader(inputStream));
49              final long insertTime = System.currentTimeMillis();
50              final PrefixTrie trie = new PrefixTrieImpl();
51              reader.lines().flatMap(s -> Arrays.stream(s.split("\s"))).forEach(trie::insert);
52              System.out.printf("Shakespeare's Sonnets inserted in %.3fms%n", (System.currentTimeMillis() - insertTime) / 1000.);
53              System.out.println(trie.size());
54              System.out.println(trie.completeWords());
55              final long findAvocado = System.nanoTime();
56              System.out.printf("Time it took to check if 'avocado' is present[%s] in Shakespeare's Sonnets %.6fns",
57                                trie.isPrefix("avocado"), (System.nanoTime() - findAvocado) / 1000_000.);
58          } catch (Exception e) {
59              e.printStackTrace();
60          }
61      }
62  
63      private static void deleteDemo(PrefixTrie trie, String toDelete) {
64          final String msg = trie.delete(toDelete) ? "true" : "false, reason: not present in tree";
65          System.out.printf("deleting '%s' -> deleted:%s%n", toDelete, msg);
66      }
67  
68      private static void startsWithDemo(PrefixTrie trie, String prefix) {
69          System.out.printf("looking up '%s' %s%n", prefix, trie.startsWith(prefix, 4));
70      }
71  }
