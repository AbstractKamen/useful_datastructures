13  /**
14   * Demo class showing PrefixTrie features and behaviour.
15   */
16  public class SuffixTrieDemo {
17      public static void main(String[] args) {
18          final Collection<String> list = List.of("apple", "apricot", "avocado",
19                                                  "ant", "app", "ants", "apps", "apis",
20                                                  "analyse", "alms", "alarm", "alarms");
21          final SuffixTrie trie = new ReversePrefixTrie();
22          list.forEach(trie::insert);
23          System.out.println(trie);
24          System.out.println(trie.prettyString());
25  
26          startsWithDemo(trie, "a");
27          startsWithDemo(trie, "ap");
28          startsWithDemo(trie, "app");
29          startsWithDemo(trie, "apple");
30          ensWithDemo(trie, "s", 4);
31          ensWithDemo(trie, "ms", 4);
32          ensWithDemo(trie, "rms", 4);
33          ensWithDemo(trie, "arm", 4);
34          ensWithDemo(trie, "arms", 4);
35          deleteDemo(trie, "appi");
36          deleteDemo(trie, "app");
37          startsWithDemo(trie, "app");
38          deleteDemo(trie, "api");
39          startsWithDemo(trie, "a");
40          System.out.println(trie.prettyString());
41          list.forEach(s -> deleteDemo(trie, s));
42          System.out.println(trie);
43          System.out.println(trie.prettyString());
44          startsWithDemo(trie, "a");
45          System.out.println("size: " + trie.size());
46          System.out.println("words: " + trie.completeWords());
47  
48          demoShakespeare();
49      }
50  
51      private static void demoShakespeare() {
52          try (final InputStream inputStream = SuffixTrieDemo.class.getClassLoader().getResourceAsStream("shakespeare.txt")) {
53              assert inputStream != null;
54              final BufferedReader reader = new BufferedReader(new InputStreamReader(inputStream));
55              final long insertTime = System.currentTimeMillis();
56              final SuffixTrie trie = new ReversePrefixTrie();
57              reader.lines().flatMap(s -> Arrays.stream(s.split("\s"))).forEach(trie::insert);
58              System.out.printf("Shakespeare's Sonnets inserted in %.3fms%n", (System.currentTimeMillis() - insertTime) / 1000.);
59              System.out.println(trie.size());
60              System.out.println(trie.completeWords());
61              final long findAvocado = System.nanoTime();
62              System.out.printf("Time it took to check if 'avocado' is present[%s] in Shakespeare's Sonnets %.6fms%n",
63                                trie.isPrefix("avocado"), (System.nanoTime() - findAvocado) / 1000_000.);
64              System.out.println("all words in Shakespeare's Sonnets");
65              ensWithDemo(trie, "ore", 100);
66              System.out.println("all words in Shakespeare's Sonnets");
67              ensWithDemo(trie, "ass", 100);
68              System.out.println("all words in Shakespeare's Sonnets");
69              ensWithDemo(trie, "ess", 100);
70          } catch (Exception e) {
71              e.printStackTrace();
72          }
73      }
74  
75      private static void deleteDemo(SuffixTrie trie, String toDelete) {
76          final String msg = trie.delete(toDelete) ? "true" : "false, reason: not present in tree";
77          System.out.printf("deleting '%s' -> deleted:%s%n", toDelete, msg);
78      }
79  
80      private static void startsWithDemo(SuffixTrie trie, String prefix) {
81          System.out.printf("starting with '%s' %s%n", prefix, trie.startsWith(prefix, 4));
82      }
83  
84      private static void ensWithDemo(SuffixTrie trie, String prefix, int limit) {
85          final Collection<String> endsWith = trie.endsWith(prefix, limit);
86          System.out.printf("ending with '%s' [%d]%s%n", prefix, endsWith.size(), endsWith);
87      }
