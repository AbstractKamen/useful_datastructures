3   /**
4    * An interface representing an adjustable binary heap data structure.
5    *
6    * <p>
7    * An adjustable heap is a binary heap that allows for efficient adjustments of element
8    * priorities.
9    * </p>
10   *
11   * <p>
12   * This interface provides methods to increase or decrease the priority of existing elements
13   * in the heap while maintaining the heap property defined by the comparator.
14   * </p>
15   *
16   * <p>
17   * Depending on the implementation the adjustable heap maintains a partial ordering, also known as a "weak ordering," among
18   * its elements. Unlike traditional binary heaps, this heap does not require a total order
19   * relation among elements. As a result, the element's relative priority may not be strictly
20   * determined with respect to other elements.
21   * </p>
22   *
23   * <p>
24   * The adjustable heap implementation is not thread-safe, and it does not allow duplicate elements.
25   * The provided comparator should be consistent with equals() and hashcode() to ensure correct behavior.
26   * </p>
27   *
28   * @param <T> The type of elements stored in the heap.
29   * @see Heap
30   * @see MergeableHeap
31   */
32  public interface AdjustableHeap<T> {
33      /**
34       * Try to increase {@code item} if it's present in this heap.
35       *
36       * @param item          to be increased
37       * @param increasedItem increasing item
38       * @return true if item is present and successfully increased
39       */
40      boolean increaseKey(T item, T increasedItem);
41  
42      /**
43       * Try to decrease {@code item} if it's present in this heap.
44       *
45       * @param item          to be decreased
46       * @param decreasedItem decreasing item
47       * @return true if item is present and successfully decreased
48       */
49      boolean decreaseKey(T item, T decreasedItem);
50  }
