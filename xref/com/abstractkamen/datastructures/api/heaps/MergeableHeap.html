3   /**
4    * <p>
5    * A mergeable heap data structure is a heap that allows for efficient merging of two heaps
6    * into a single heap. Elements stored in the heap must be comparable, and the provided
7    * comparator defines the ordering of the elements.
8    * </p>
9    *
10   * <p>
11   * The merge operation combines both heaps into a new heap with all elements from
12   * both heaps, maintaining the heap property defined by the comparator of the heap
13   * invoking {@link #mergeWith(Heap)}.
14   * </p>
15   *
16   * @param <T> The type of elements stored in the heap.
17   * @see Heap
18   * @see AdjustableHeap
19   */
20  public interface MergeableHeap<T> {
21      /**
22       * Merges this heap with the other using {@code this.}{@link Heap#comparator()}. The heaps will be merged regardless of {@code other}
23       * 's order.
24       * Depending on the implementation this may have an impact on performance.
25       *
26       * <pre>
27       * Example:
28       *     {@code MergeableHeap<Heap<Integer>> a = new SomeMergeableHeapImpl(Integer::compare);}
29       *     // can be merged with
30       *     {@code MergeableHeap<Heap<Integer>> b = new SomeMergeableHeapImpl(Comparator.reverseOrder(Integer::compare));}
31       *     // different implementations will have different time complexity
32       *     a.mergeWith(b);
33       * </pre>
34       *
35       * @param other heap
36       * @return {@code this} merged with other
37       * @throws ClassCastException if {@code other} not the same instance as {@code this}
38       */
39      Heap<T> mergeWith(Heap<T> other);
40  }
