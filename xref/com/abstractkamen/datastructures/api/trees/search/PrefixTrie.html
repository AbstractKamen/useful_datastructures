1   package com.abstractkamen.datastructures.api.trees.search;
2   
3   import java.util.Collection;
4   
5   /**
6    * The PrefixTrie can perform fast prefix matching. By traversing the trie based on the characters in
7    * the prefix being searched, it is possible to find all strings in the Trie that have the given prefix.
8    */
9   public interface PrefixTrie {
10  
11      /**
12       * Try to find strings which start with prefix. Matching strings are always in lexicographical order.
13       *
14       * @param prefix to look for
15       * @param limit maximum number of found strings
16       * @return collection of words with the same prefix
17       */
18      Collection<String> startsWith(String prefix, int limit);
19  
20      /**
21       * Checks if a word exists in this trie.
22       *
23       * @param string to check
24       * @return true if word is in this trie
25       */
26      boolean contains(String string);
27  
28      /**
29       * Checks if prefix is present in this trie.
30       *
31       * @param prefix to check
32       * @return true if prefix exists
33       */
34      boolean isPrefix(String prefix);
35  
36      /**
37       * Inserts a word into the trie. Duplicate words will be ignored.
38       *
39       * @param string word to be inserted
40       * @return true if the word was inserted false if it already existed
41       */
42      boolean insert(String string);
43  
44      /**
45       * Deletes a word from the trie. The number of complete words will be decremented by one and the size of the trie will shrink
46       * by the number of unique characters and their placements in the deleted word if they aren't used by a different word.
47       *
48       * @param string to be deleted
49       * @return true if word is successfully deleted false if word was not in the trie
50       */
51      boolean delete(String string);
52  
53      /**
54       * The number of nodes in this trie.
55       * @return number of nodes
56       */
57      int size();
58  
59      /**
60       * The number of complete words in this trie.
61       *
62       * @return number of complete words
63       */
64      int completeWords();
65  
66      /**
67       * A detailed string representation of the structure of this trie.
68       *
69       * @return detailed string
70       */
71      String prettyString();
72  
73  }
