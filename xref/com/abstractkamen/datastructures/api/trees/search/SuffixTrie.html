1   package com.abstractkamen.datastructures.api.trees.search;
2   
3   import java.util.Collection;
4   
5   /**
6    * An extension of the prefix trie. This structure can perform fast suffix matches at the cost of a structure double the size of a
7    * PrefixTrie.
8    */
9   public interface SuffixTrie extends PrefixTrie {
10  
11      /**
12       * Try to find strings which end with suffix. Matching strings are always in arbitrary order.
13       *
14       * @param suffix to look for
15       * @param limit maximum number of found strings
16       * @return collection of words with the same suffix
17       */
18      Collection<String> endsWith(String suffix, int limit);
19  
20      /**
21       * Checks if suffix is present in this trie.
22       *
23       * @param suffix to check
24       * @return true if suffix exists
25       */
26      boolean isSuffix(String suffix);
27  }
