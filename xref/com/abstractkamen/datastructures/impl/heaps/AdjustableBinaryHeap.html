7   /**
8    * An implementation of an adjustable binary heap data structure.
9    *
10   * @param <T> The type of elements stored in the heap, must be comparable.
11   * @see Heap
12   * @see MergeableHeap
13   * @see AdjustableHeap
14   */
15  public class AdjustableBinaryHeap<T> extends BinaryHeap<T> implements Heap<T>, MergeableHeap<T>, AdjustableHeap<T> {
16  
17      /**
18       * Create an {@code AdjustableBinaryHeap<T>} with a custom comparator
19       *
20       * @param comparator custom comparator
21       */
22      public AdjustableBinaryHeap(Comparator<T> comparator) {
23          super(comparator, DEFAULT_CAPACITY);
24      }
25  
26      /**
27       * Create an {@code AdjustableBinaryHeap<T>} with a custom comparator
28       *
29       * @param capacity   initial capacity
30       * @param comparator custom comparator
31       */
32      public AdjustableBinaryHeap(Comparator<T> comparator, int capacity) {
33          super(comparator, capacity);
34      }
35  
36      /**
37       * Create an {@code AdjustableBinaryHeap<T>} with natural order comparator in a type safe way.
38       *
39       * @param <T> comparable type
40       */
41      public static <T extends Comparable<T>> AdjustableBinaryHeap<T> createComparable() {
42          final Comparator<T> c = Comparable::compareTo;
43          return new AdjustableBinaryHeap<>(c);
44      }
45  
46      /**
47       * Create an {@code AdjustableBinaryHeap<T>} with natural order comparator in a type safe way.
48       *
49       * @param capacity initial capacity
50       * @param <T>      comparable type
51       */
52      public static <T extends Comparable<T>> AdjustableBinaryHeap<T> createComparable(int capacity) {
53          final Comparator<T> c = Comparable::compareTo;
54          return new AdjustableBinaryHeap<>(c);
55      }
56  
57      @Override
58      public boolean increaseKey(T item, T increasedItem) {
59          return false;
60      }
61  
62      @Override
63      public boolean decreaseKey(T item, T decreasedItem) {
64          final int compare = comparator().compare(item, decreasedItem);
65          if (compare < 0) return false;
66          final Object[] items = getItems();
67          int i = -1;
68          for (int j = 0; j < size(); j++) {
69              if (item == items[j]) {
70                  i = j;
71                  break;
72              }
73          }
74          if (i > -1 && compare > 0) {
75              items[i] = decreasedItem;
76              heapifyUp(items, comparator(), i);
77          } else {
78              push(decreasedItem);
79          }
80          return true;
81      }
82  
83      @Override
84      public AdjustableBinaryHeap<T> mergeWith(Heap<T> other) {
85          return (AdjustableBinaryHeap<T>) super.mergeWith(other);
86      }
87  }
