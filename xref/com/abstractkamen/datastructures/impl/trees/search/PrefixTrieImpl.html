10      private static final String LINE_SEPARATOR = System.lineSeparator();
11      private final NaiveTrieNode root;
12      private int completeWords;
13      private int size;

15      public PrefixTrieImpl() {
16          this.root = new NaiveTrieNode();
17      }

19      @Override
20      public Collection<String> startsWith(String prefix, int limit) {
21          final List<NaiveTrieNode> leaves = new ArrayList<>();
22          search(this.root, prefix, 0, leaves, limit);
23          return getWords(leaves);
24      }
25  
26      @Override
27      public boolean isPrefix(String substring) {
28          return search(this.root, substring, 0, new ArrayList<>(), 1);
29      }
30  
31      @Override
32      public boolean insert(String string) {
33          if (string == null || string.isEmpty()) return false;
34          final NaiveTrieNode inserted = insert(string, this.root, 0);
35          if (inserted.isWord) {
36              // was inserted before
37              return false;
38          }
39          this.completeWords++;
40          inserted.setIsWord(true);
41          return true;
42      }
43  
44      @Override
45      public boolean delete(String string) {
46          return delete(this.root, string, 0);
47      }
48  
49      @Override
50      public int size() {
51          return size;
52      }
53  
54      @Override
55      public int completeWords() {
56          return completeWords;
57      }
58  
59      @Override
60      public String toString() {
61          return startsWith("", completeWords).toString();
62      }
63  
64      @Override
65      public boolean contains(String string) {
66          final List<NaiveTrieNode> result = new ArrayList<>();
67          search(this.root, string, 0, result, 1);
68          return !result.isEmpty();
69      }

71      @Override
72      public String prettyString() {
73          if (this.root.children.isEmpty()) {
74              return "empty tree";
75          } else {
76              final StringBuilder sb = new StringBuilder("NaiveTrie total nodes: ")
77                  .append(size).append(LINE_SEPARATOR)
78                  .append("total complete words:")
79                  .append(completeWords).append(LINE_SEPARATOR);
80              visitAllNodes(0, this.root, ">>>", sb);
81              return sb.toString();
82          }
83      }

85      private void visitAllNodes(int size, NaiveTrieNode node, String prefix, StringBuilder visitor) {
86          final String pointer = "├─► ";
87          final String hookPointer = "└─► ";
88          final String pointerConnection = "│";
89          // depth-first search
90          // visit current
91          if (node != this.root) {
92              if (size > 1)
93                  visitor.append(prefix).append(pointer);
94              else {
95                  visitor.append(prefix).append(hookPointer);
96              }
97              visitor.append((char) node.c);
98          }
99          // if current is word append the whole word to the current node
100         if (node.isWord) {
101             NaiveTrieNode c = node;
102             visitor.append(": (");
103             final int l = visitor.length();
104             while (c != this.root) {
105                 visitor.insert(l, (char) c.c);
106                 c = c.parent;
107             }
108             visitor.append(")");
109         }
110         visitor.append(LINE_SEPARATOR);
111         final String nextPrefix;
112         if (node.parent != null && size > 1) {
113             nextPrefix = prefix + pointerConnection + " ".repeat(pointer.length());
114         } else if (node != this.root) {
115             nextPrefix = prefix + " ".repeat(pointer.length());
116         } else {
117             nextPrefix = prefix;
118         }
119         // visit children
120         int s = node.children.size();
121         for (NaiveTrieNode child : node.children.values()) {
122             visitAllNodes(s--, child, nextPrefix, visitor);
123         }
124     }

126     private NaiveTrieNode insert(String word, NaiveTrieNode root, int i) {
127         if (i == word.length()) {
128             return root;
129         } else {
130             final int c = word.charAt(i);
131             NaiveTrieNode child = root.children.get(c);
132             if (child == null) {
133                 child = new NaiveTrieNode(c, root);
134                 size++;
135                 root.children.put(c, child);
136             }
137             return insert(word, child, i + 1);
138         }
139     }

141     private Collection<String> getWords(List<NaiveTrieNode> leaves) {
142         return leaves.stream().map(l -> {
143             final StringBuilder sb = new StringBuilder();
144             while (l != this.root) {
145                 sb.insert(0, (char) l.c);
146                 l = l.parent;
147             }
148             return sb.toString();
149         }).collect(Collectors.toList());
150     }

152     /**
153      * Searching method for this trie. Will walk through all characters in needle and find a possible path of nodes for them. If the path
154      * exists return true.
155      * @param n current node
156      * @param needle string
157      * @param i offset
158      * @param result string storage
159      * @param resultLimit storage limit
160      * @return true if a needle path exists
161      */
162     private boolean search(NaiveTrieNode n, String needle, int i, Collection<NaiveTrieNode> result, int resultLimit) {
163         if (n == null) {
164             // we fell off the tree
165             return false;
166         } else if (result.size() >= resultLimit) {
167             //we hit the limit and must return, but we haven't fallen off
168             return true;
169         }
170         if (n.isWord && i >= needle.length()) {
171             result.add(n);
172         }
173         if (i >= needle.length()) {
174             // assume always hit
175             for (NaiveTrieNode c : n.children.values()) {
176                 search(c, needle, i + 1, result, resultLimit);
177             }
178             return true;
179         } else {
180             final int c = needle.charAt(i);
181             // go deeper
182             return search(n.children.get(c), needle, i + 1, result, resultLimit);
183         }
184     }

186     private boolean delete(NaiveTrieNode n, String word, int i) {
187         if (n == null || i > word.length()) {
188             return false;
189         }
190         if (n.isWord && i == word.length()) {
191             completeWords--;
192             if (!n.children.isEmpty()) {
193                 n.setIsWord(false);
194             } else {
195                 removeNode(n.parent, n.c);
196             }
197             return true;
198         } else if (i < word.length()) {
199             final int c = word.charAt(i);
200             return delete(n.children.get(c), word, i + 1);
201         } else {
202             return false;
203         }
204     }
205 
206     private void removeNode(NaiveTrieNode parent, int c) {
207         if (parent != null) {
208             parent.children.remove(c);
209             if (parent.children.isEmpty()) {
210                 removeNode(parent.parent, parent.c);
211             }
212             size--;
213         }
214     }
